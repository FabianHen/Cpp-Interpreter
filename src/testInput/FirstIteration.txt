class A{
public:
    int var;
    virtual void hello(int a){
        print_char('A');
        print_char('\n');
    }
};

class B : public A{
public:
    int varB;
    void hello(){
        print_char('B');
        print_char('\n');
    }
};

B b;
b.var = 5;
b.varB = 10;
print_int(b.var);
print_int(b.varB);
